Теперь будем делать соединения. Берем каждый SMILES и ищем в нем висящую азидную группу. Выствим метки на азотах. Пусть на самом крайнем азоте метка 3, а далее на последующих 4 и 5. Потом надо проверить правильная ли азидная группа попалась: соответствие зарядов и связей. Когда все оказалось правильно и такая группа найдена, необходимо соединить исправленный ибупрофен с остатком рассматриваемой молекулы по схеме https://en.wikipedia.org/wiki/Click_chemistry. Образуем кольцо: C1-N=N=N-C=C1. Где мы знаем, что висячий углерод имел метку 2, поэтому его необходимо соединить с азотом номер 3, а углерод 1 с азотом 5. Теперь надо поменять еще тип связи, потому что не всегда получается так, как нужно.

def check_azid_group(molecule, group):
    charges = list(map(lambda x: x.GetFormalCharge(), group))
    first_bond_type = molecule.GetBondBetweenAtoms(group[0].GetIdx(), group[1].GetIdx()).GetBondType()
    second_bond_type = molecule.GetBondBetweenAtoms(group[1].GetIdx(), group[2].GetIdx()).GetBondType()
    if charges == [-1, 1, 0]:
        return first_bond_type == Chem.rdchem.BondType.DOUBLE and second_bond_type == Chem.rdchem.BondType.DOUBLE
    elif charges == [0, 1, -1]:
        return first_bond_type == Chem.rdchem.BondType.TRIPLE and second_bond_type == Chem.rdchem.BondType.SINGLE
    elif charges == [1, 0, -1]:
        return first_bond_type == Chem.rdchem.BondType.DOUBLE and second_bond_type == Chem.rdchem.BondType.SINGLE
    return False

def make_compaund(ibuprophen, mol):
    product = Chem.CombineMols(ibuprophen, mol)
    reaction_site = [0] * 5
    for atom in product.GetAtoms():
        if atom.GetAtomMapNum() > 0:
            reaction_site[atom.GetAtomMapNum() - 1] = atom.GetIdx()
    for i in range(2, 5):
        product.GetAtomWithIdx(reaction_site[i]).SetFormalCharge(0)
    edproduct = Chem.EditableMol(product)
    edproduct.AddBond(reaction_site[0], reaction_site[4], order=Chem.rdchem.BondType.SINGLE)
    edproduct.AddBond(reaction_site[1], reaction_site[2], order=Chem.rdchem.BondType.SINGLE)
    product = edproduct.GetMol()
    product.GetBondBetweenAtoms(reaction_site[0], reaction_site[1]).SetBondType(Chem.rdchem.BondType.DOUBLE)
    product.GetBondBetweenAtoms(reaction_site[2], reaction_site[3]).SetBondType(Chem.rdchem.BondType.DOUBLE)
    product.GetBondBetweenAtoms(reaction_site[3], reaction_site[4]).SetBondType(Chem.rdchem.BondType.SINGLE)
    
    for atom in product.GetAtoms():
        atom.SetAtomMapNum(0)
    
    return product
