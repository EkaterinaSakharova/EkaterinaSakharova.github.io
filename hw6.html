# Зададим параметры окружения
env = modeller.environ()
env.io.hetatm = True  # New objects will read HETATM records from PDB by default 

# Это я взяла UniProtKB - P61626 (LYSC_HUMAN)
alignm = modeller.alignment(env)
alignm.append(file='P61626.fasta', align_codes='all', alignment_format='FASTA')

# модель номер раз
mdl = modeller.model(env, file='1lmp.pdb', model_segment=('FIRST:'+'A', 'LAST:'+'A'))
# добавим в выравнивание
alignm.append_model(mdl, atom_files='1lmp.pdb', align_codes='1lmp')
# идентификатор первого белка
alignm[0].code = 'bad_protein'

# Делаем выравнивание и сохраняем:
alignm.salign()
alignm.write(file='bad_protein.ali', alignment_format='PIR')
s = alignm[0]
pdb = alignm[1]
print s.code, pdb.code

# Создаем первый объект automodel
a = modeller.automodel.automodel(env, alnfile='bad_protein.ali', knowns= pdb.code , sequence = s.code )
a.name='mod'+s.code
a.starting_model = 1
a.ending_model = 1
a.make()

Лиганда в этой структуре нет. Это можно увидеть и в файле выравнивания. Лиганд можно приписать к выравниванию, как точки. Такую вещь можно и в документации найти, например, https://salilab.org/modeller/9v7/manual/node18.html#SECTION:model-ligand
