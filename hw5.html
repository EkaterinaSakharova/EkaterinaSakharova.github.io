def energy_calculation(way1,N):
    
    ENERGY = []
    for i in range(0,N):
        way = way1+str(i)+'.inp'
        p = subprocess.Popen(way, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        out=p.communicate()[0]
        out.splitlines()

        index = out.find('FINAL SINGLE POINT ENERGY')
        l = len('FINAL SINGLE POINT ENERGY  ')
        fl = 1
        while (fl==1):
            if out[l+index]==' ' : l = l+1
            else: fl = 2
        start = index+l        
        while (fl == 2):
            if out[l+index]!='\n' : l = l+1
            else: fl = 3
        finish = index+l
        ENERGY.append(float(out[start:finish]))
    return ENERGY

def plot_energy(x_o, y_o, col):
    
    # function is  f(x)=k(b-x)^2 + a
    fitfunc = lambda p, x: p[0]*pow(p[1]-x,2) + p[2] # Target function
    errfunc = lambda p, x, y: fitfunc(p, x) - y # Error function

    p0 = [1,1, -79] # Initial guess for the parameters
    p1, success = optimize.leastsq(errfunc, p0[:], args=(x_o, y_o))
    print "Optimized params:", p1

    # Plot
    plt.plot(x_o, y_o, "ro", x_o,fitfunc(p1,x_o),"r-",c=col,alpha=0.5)
    plt.show()
